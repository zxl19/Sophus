Overview¶

Specialization of Eigen::Map for SE2 const; derived from SE2Base. More…

#include <se2.hpp>

template <class Scalar_, int Options>
class Map<Sophus::SE2<Scalar_> const, Options>: public Sophus::SE2Base {
public:
    // typedefs

    using Base = Sophus::SE2Base<Map<Sophus::SE2<Scalar_> const, Options>> ;
    using Scalar = Scalar_ ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar const* coeffs);

    // methods

    Map<Sophus::SO2<Scalar> const, Options> const& so2() const;
    Map<Sophus::Vector2<Scalar> const, Options> const& translation() const;

protected:
    // fields

    const Map<Sophus::SO2<Scalar> const, Options> so2_;
    const Map<Sophus::Vector2<Scalar> const, Options> translation_;
};
