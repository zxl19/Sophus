#include <sim2.hpp>

template <class Scalar_, int Options>
class Map<Sophus::Sim2<Scalar_>, Options>: public Sophus::Sim2Base {
public:
    // typedefs

    using Base = Sophus::Sim2Base<Map<Sophus::Sim2<Scalar_>, Options>> ;
    using Scalar = Scalar_ ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar* coeffs);

    // methods

    Map<Sophus::RxSO2<Scalar>, Options>& rxso2();
    Map<Sophus::RxSO2<Scalar>, Options> const& rxso2() const;
    Map<Sophus::Vector2<Scalar>, Options>& translation();
    Map<Sophus::Vector2<Scalar>, Options> const& translation() const;

protected:
    // fields

    Map<Sophus::RxSO2<Scalar>, Options> rxso2_;
    Map<Sophus::Vector2<Scalar>, Options> translation_;
};

Detailed Documentation¶

Specialization of Eigen::Map for Sim2; derived from Sim2Base.

Allows us to wrap Sim2 objects around POD array.
