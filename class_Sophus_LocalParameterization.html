template class Sophus::LocalParameterization¶

Overview¶

Templated local parameterization for LieGroup [with implemented LieGroup::Dx_this_mul_exp_x_at_0() ]. More…

#include <ceres_local_parameterization.hpp>

template <template<typename, int=0> class LieGroup>
class LocalParameterization: public LocalParameterization {
public:
    // typedefs

    using LieGroupd = LieGroup<double> ;
    using Tangent = typename LieGroupd::Tangent ;
    using TangentMap = typename Sophus::Mapper<Tangent>::ConstMap ;

    // fields

    static constexpr int DoF = LieGroupd::DoF;
    static constexpr int num_parameters = LieGroupd::num_parameters;

    // methods

    bool Plus(double const* T_raw, double const* delta_raw, double* T_plus_delta_raw) const;
    bool ComputeJacobian(double const* T_raw, double* jacobian_raw) const;
    int GlobalSize() const;
    int LocalSize() const;
};
